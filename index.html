find_mode<-function(x){
	values<-unique(x)
	counts<-tabulate(match(x,values))
	mode_index<-which.max(counts)
	mode<-values[mode_index]
	return(mode)
}
a<-c(1,2,3,3,4,5,3,2,1,1)
mode<-find_mode(a)
print(mode)

library(readxl)
data<-c(1,2,2,3,3,3,4,4,4,4,5,5,8,8,8,8)

skew<-skewness(data)
test(x=max(data)-1,y=3,labels=paste("Skewness",round(skew,2))
